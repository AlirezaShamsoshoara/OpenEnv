models.py Data Types 
from openenv import Action, Observation, State

class HackAction(Action):
    command: str              # e.g. "nmap -sV target"
    target: str = ""

class HackObservation(Observation):
    output: str = ""          # command output
    flags_found: list[str] = []

class HackState(State):
    vulnerabilities_found: int = 0
    total_flags: int = 5
server/hack_environment.py Core Logic 
from openenv import MCPEnvironment
from mcp.server.fastmcp import FastMCP

class HackEnvironment(MCPEnvironment):
    def __init__(self):
        mcp = FastMCP("hack_env")
        @mcp.tool
        def scan_port(target: str, port: int) -> str: ...
        @mcp.tool
        def exploit_vuln(vuln_id: str) -> dict: ...
        super().__init__(mcp)

    def reset(self, **kw) -> HackObservation: ...
    def step(self, action, **kw) -> HackObservation: ...
server/app.py Entry Point 
from openenv import create_app
from .hack_environment import HackEnvironment
from ..models import HackAction, HackObservation

app = create_app(HackEnvironment, HackAction, HackObservation, env_name="hack_env")
