from uuid import uuid4
from openenv import MCPEnvironment
from mcp.server.fastmcp import FastMCP
from ..models import HackObservation, HackState

class HackEnvironment(MCPEnvironment):
    # Hidden CTF flags the agent must discover
    FLAGS = {"web_vuln", "sql_inject", "priv_esc",
             "exposed_api", "weak_creds"}

    def __init__(self):
        mcp = FastMCP("hack_env")

        @mcp.tool
        def scan_port(host: str, port: int) -> str:
            """Scan a port on the target host."""
            if port == 8080:
                self._found.add("exposed_api")
                return "OPEN - HTTP API (no auth)"
            return f"Port {port}: closed"

        @mcp.tool
        def check_sql(url: str, payload: str) -> str:
            """Test SQL injection on endpoint."""
            if "' OR" in payload:
                self._found.add("sql_inject")
                return "VULNERABLE: SQL injection!"
            return "No vulnerability found"

        @mcp.tool
        def brute_login(user: str, pwd: str) -> str:
            """Attempt login with credentials."""
            if pwd in ["admin", "password", "123456"]:
                self._found.add("weak_creds")
                return "LOGIN SUCCESS (weak password!)"
            return "Login failed"

        super().__init__(mcp)
        self._state = HackState()
        self._found = set()

    def reset(self, seed=None, episode_id=None,
              **kwargs) -> HackObservation:
        self._found = set()
        self._state = HackState(
            episode_id=episode_id or str(uuid4()),
            step_count=0,
            total_flags=len(self.FLAGS)
        )
        return HackObservation(
            output="Target system ready. Find 5 flags.",
            done=False, reward=0.0
        )

    def step(self, action, timeout_s=None,
             **kwargs) -> HackObservation:
        self._state.step_count += 1
        obs = super().step(action, timeout_s)

        # Compute reward: +1 per new flag
        new_flags = len(self._found)
        reward = new_flags / len(self.FLAGS)
        done = new_flags == len(self.FLAGS)

        self._state.vulnerabilities_found = new_flags
        return HackObservation(
            output=obs.output,
            flags_found=list(self._found),
            reward=reward, done=done
        )

    @property
    def state(self) -> HackState:
        return self._state
