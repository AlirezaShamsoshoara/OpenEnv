Creating a Client Environment

Type-safe client that connects to any OpenEnv server via WebSocket

client.py 
from openenv import EnvClient, StepResult
from .models import (
    HackAction, HackObservation, HackState
)

class HackEnv(
    EnvClient[HackAction, HackObservation, HackState]
):
    """Client for the HackEnv CTF environment."""

    def _step_payload(self, action: HackAction):
        # Serialize action to JSON for WebSocket
        return action.model_dump()

    def _parse_result(self, payload):
        # Deserialize server response
        obs = HackObservation(**payload)
        return StepResult(
            observation=obs,
            reward=obs.reward,
            done=obs.done
        )

    def _parse_state(self, payload):
        return HackState(**payload)

# --- Usage: Async (recommended) ---
async with HackEnv(
    base_url="http://localhost:8000"
) as env:
    obs = await env.reset()
    result = await env.step(action)

# --- Usage: Sync wrapper ---
with HackEnv(base_url="...").sync() as env:
    obs = env.reset()
    result = env.step(action)

Generic Type Safety

EnvClient[ActT, ObsT, StateT] ensures compile-time type checking. All validated via Pydantic.

Three Abstract Methods

Implement _step_payload (serialize), _parse_result (deserialize obs), and _parse_state (deserialize state).
