run_hackenv.py 
import asyncio
from hack_env import HackEnv
from openenv import MCPToolClient

async def main():
    # Option 1: Typed client
    async with HackEnv(
        base_url="http://localhost:8000"
    ) as env:

        # Reset - start new episode
        obs = await env.reset()
        print(obs.output)

        # Step 1: Scan for open ports
        result = await env.step(
            CallToolAction(
                name="scan_port",
                arguments={
                    "host": "target.local",
                    "port": 8080
                }
            )
        )
        print(result.observation.output)
        print(f"Reward: {result.reward}")

        # Step 2: Test SQL injection
        result = await env.step(
            CallToolAction(
                name="check_sql",
                arguments={
                    "url": "/api/users",
                    "payload": "' OR 1=1--"
                }
            )
        )

        # Check state anytime
        state = await env.state()
        print(f"Found {state.vulnerabilities_found}/{state.total_flags}")

asyncio.run(main())
